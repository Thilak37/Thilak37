#!/bin/bash

# Step 1: Initialize a local Git repository
git init

# Step 2: Add files to the staging area
git add .

# Step 3: Commit changes Name can be provided within double quotes
git commit -m "Add new feature"

# Step 4: Add a remote repository (if not already added)
git remote add origin <remote-url>

# Step 5: Push changes to the remote repository
git push origin master
